def fetch_and_format_user_agents(url):
    try:
        response = requests.get(url)
        if response.status_code == 200:
            user_agents = response.text.split('\n')
            return [f"'{ua.strip()}'" for ua in user_agents if ua.strip()]
    except:
        pass
    return []
